# vuepress-sitemap

# 简介

根据所有 page 生成标签和分类 map,注入全局 mixins

# 使用

加入插件

  // enhanceApp.js
  import VueSiteMap from './plugins/vuepress-sitemap'

  export default ({
    Vue,
    options,
    router,
    siteData
  }) => {
    const { pages } = siteData
    Vue.use(VueSiteMap, {
      pages
    })
  }
